import { useContext, useEffect, useState } from 'react';
import { Routes, Route, Navigate } from 'react-router-dom';
import { observer } from 'mobx-react-lite';

import { Context } from '@/context';
import ProtectedRoute from '@/components/ProtectedRoute.jsx';
import PublicOnlyRoute from '@/components/PublicOnlyRoute.jsx';
import { publicRoutes, authRoutes } from '@/routes';
import '@/styles/theme.css';

const App = observer(() => {
  const { user } = useContext(Context);
  const [loading, setLoading] = useState(true);

  useEffect(() => { setTimeout(() => setLoading(false), 0); }, []);
  if (loading) return <div style={{ padding: 24 }}>–ó–∞–≥—Ä—É–∑–∫–∞‚Ä¶</div>;

  return (
    <Routes>
      {/* –ø—É–±–ª–∏—á–Ω—ã–µ */}
      {publicRoutes.map(({ path, element }) => (
        <Route
          key={`pub-${path}`}
          path={path}
          element={<PublicOnlyRoute>{element}</PublicOnlyRoute>}
        />
      ))}

      {/* –ø—Ä–∏–≤–∞—Ç–Ω—ã–µ (–≤–∫–ª—é—á–∞—è nested) */}
      {authRoutes.map(({ path, element, children, adminOnly }) => (
        <Route
          key={`auth-${path || 'root'}`}
          path={path}
          element={
            <ProtectedRoute>
              {adminOnly && !user.isAdmin ? <Navigate to="/dashboard" replace /> : element}
            </ProtectedRoute>
          }
        >
          {Array.isArray(children) && children.map((c) => (
            <Route
              key={`${path}-${c.path || 'index'}`}
              path={c.path}
              element={
                <ProtectedRoute>
                  {c.adminOnly && !user.isAdmin ? <Navigate to="/dashboard" replace /> : c.element}
                </ProtectedRoute>
              }
            />
          ))}
        </Route>
      ))}

      {/* —Ñ–æ–ª–ª–±—ç–∫ */}
      <Route path="*" element={<Navigate to={user.isAuth ? '/dashboard' : '/login'} replace />} />
    </Routes>
  );
});

export default App;

  const [show, setShow] = useState(true);
  useEffect(() => {
    const id = setTimeout(()=>setShow(false), duration);
    return () => clearTimeout(id);
  }, [duration]);

  if (!show) return null;
  return (
    <div className={`${styles.toast} ${styles[type]}`}>
      {message}
    </div>
  );
}

  try {
    if (!token) return null;
    const payload = token.split('.')[1];
    const json = JSON.parse(decodeURIComponent(escape(atob(payload))));
    // –æ–∂–∏–¥–∞–µ–º –ø–æ–ª—è sub/email/roles ‚Äî –ø–æ–¥—Å—Ç—Ä–æ–π –ø—Ä–∏ –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ—Å—Ç–∏

      id: json.userId || json.id || null,
      email: json.email || json.sub || null,
      roles: Array.isArray(json.roles) ? json.roles : (json.authorities || []),
      raw: json,
    };
  } catch {

  }

    const status = error?.response?.status;

    if (status === 401 && !original._retry) {

        return new Promise((resolve, reject) => {
          pendingQueue.push({
            resolve: (token) => {

              resolve($authHost(original));
            },
            reject,

        });
      }

import { makeAutoObservable } from 'mobx';
import { saveToken, getToken, removeToken, parseUserFromToken, $host } from '@/http';
import { registerApi, loginApi, logoutApi } from '@/http/userApi';

export default class UserStore {
  user = null;
  isAuth = false;

  constructor() { makeAutoObservable(this); }

  setAuth(v) { this.isAuth = v; }
  setUser(u) { this.user = u; }

  get isAdmin() {
    const roles = this.user?.roles || [];
    return roles.includes('ROLE_ADMIN');
  }

  // –≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø—Ä–∏ —Å—Ç–∞—Ä—Ç–µ –ø—Ä–∏–ª–æ–∂–µ–Ω–∏—è
  async checkOnStart() {
    const token = getToken();
    if (token) {
      const u = parseUserFromToken(token);
      if (u?.email) { this.setUser(u); this.setAuth(true); }
      else removeToken();
    }
    // —Ç–∏—Ö–∞—è –ø–æ–ø—ã—Ç–∫–∞ refresh ‚Äî –æ–∫, –µ—Å–ª–∏ 403/401
    try {
      const resp = await $host.post('/api/v1/auth/refresh', {}, { withCredentials: true });
      const newToken = resp?.data?.accessToken || resp?.data?.token;
      if (newToken) {
        saveToken(newToken);
        const u2 = parseUserFromToken(newToken);
        this.setUser(u2);
        this.setAuth(true);
      }
    } catch (_) {}
  }

  // –†–ï–ì–ò–°–¢–†–ê–¶–ò–Ø
  async register({ email, password, fullName }) {
    const { data } = await registerApi({ email, password, fullName });
    const token = data?.accessToken || data?.token;
    if (token) {
      saveToken(token);
      const u = parseUserFromToken(token);
      this.setUser(u);
      this.setAuth(true);
    }
    return data;
  }

  // –õ–û–ì–ò–ù
  async login({ email, password }) {
    const { data } = await loginApi({ email, password });
    const token = data?.accessToken || data?.token;
    if (token) {
      saveToken(token);
      const u = parseUserFromToken(token);
      this.setUser(u);
      this.setAuth(true);
    }
    return data;
  }

  // –í–´–•–û–î
  async logout() {
    try { await logoutApi(); } catch (_) {}
    removeToken();
    this.setUser(null);
    this.setAuth(false);
  }
}
