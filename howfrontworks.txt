/dashboard/passes → frontend/src/pages/user/MyPasses.jsx
/dashboard/profile → frontend/src/pages/user/Profile.jsx
/dashboard/admin → frontend/src/pages/Admin/Home/AdminHome.jsx
/dashboard/admin/queue → frontend/src/pages/Admin/Queue/AdminQueue.jsx

/dashboard/admin/employees → frontend/src/pages/Admin/Employees.jsx

Дополнительно:
